function sleep(millis) {
    let cnt = 0;
    const interval = setInterval(function () {
        console.log(cnt);
        cnt++;
        if (cnt === 5) {
            clearInterval(interval);
            console.log('end');
        }
    key('sentence_container', 65, false, 'key "あ"');
    }, millis);
}

function click(id, msg) {
    const clickElem = document.getElementById(id);
    console.log(clickElem);

    if (document.all) {
        clickElem.fireEvent('onclick');
    } else {
        console.log('click');
        const event = document.createEvent('MouseEvents');
        event.initEvent('click', false, true);
        clickElem.dispatchEvent(event);
    }

    message(msg);
}

function test_click() {
    click('btn_open_file', 'push file button!');
}

function test_key() {
    key('sentence_container', 65, false, 'key auto');
    return 'end test_key';
}

function key(id, key_code, with_shift, msg) {
    const type = 'keydown';
    const elem = document.getElementById(id);
    console.log(elem);
    if (document.createEventObject) {
        const f_event = document.createEventObject();
        f_event.keyCode = key_code;
        f_event.shiftKey = with_shift;
        elem.fireEvent('on' + type, f_event);
    } else {
        const f_event = document.createEvent('Event');
        f_event.initEvent(type, true, false);
        f_event.keyCode = key_code;
        f_event.shiftKey = with_shift;
        elem.dispatchEvent(f_event);
        console.log(f_event);
    }
    message(msg);
}

function message(msg) {
    const eMsg = document.getElementById('test_message');
    eMsg.textContent = msg;
}
